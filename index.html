											you can position it according to your needs.
										-->
										</div>
										<script src="path/to/folder/notification.min.js"></script>
										<script>
											Notification.configs({
												stacking: true, // Default set to true,
												// Removes notifications that are out of viewport, and adds a show all button
											});
											Notification.create({
												header: "My header",
												body: "Hey, I'm a notification üòç",
												footer: "My footer",
												timeout: 2, // when set to Infinity, notification will never auto close.  
												pause: true, // Default set to false, pauses timeout of notification 
												timebar: true // Default set to false, Adds a progress bar showing timeout 
												dismiss: true, // Default set to false, Adds a close button inside notification
											});
										</script>
									</body>
									</html>
